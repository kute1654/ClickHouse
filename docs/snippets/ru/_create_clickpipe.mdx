import GCSSDataSource from '/snippets/es/clickpipes/object-storage/google-cloud-storage/_1-data-source.mdx';

import S3Connection from '/snippets/es/clickpipes/object-storage/amazon-s3/_2-connection.mdx';
import GCSConnection from '/snippets/es/clickpipes/object-storage/google-cloud-storage/_2-connection.mdx';

<Steps>
  <Step>
    ## Выберите источник данных

    **1.** В ClickHouse Cloud выберите **Data sources** в главном навигационном меню и нажмите **Create ClickPipe**.

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_step0.webp" alt="Выбор импорта" size="lg" border />

    {props.provider === 's3' && <S3DataSource />}

    {props.provider === 'gcs' && <GCSSDataSource />}
  </Step>

  <Step>
    ## Настройте подключение ClickPipe

    **1.** Чтобы настроить новый ClickPipe, необходимо указать параметры подключения и аутентификации для вашего сервиса Объектного хранилища.

    {props.provider === 's3' && <S3Connection />}

    {props.provider === 'gcs' && <GCSConnection />}

    **2.** Нажмите **Incoming data**. ClickPipes получит метаданные из вашего бакета для следующего шага.
  </Step>

  <Step>
    ## Выберите формат данных

    В интерфейсе отобразится список файлов в указанном бакете.
    Выберите формат данных (сейчас поддерживается только часть форматов ClickHouse), а также укажите, нужно ли включить непрерывный приём.
    Подробнее см. в разделе &quot;continuous ingest&quot; на странице обзора.

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_step3_object_storage.webp" alt="Задать формат данных и topic" size="lg" border />
  </Step>

  <Step>
    ## Настройте таблицу, схему и параметры

    На следующем шаге вы можете выбрать, выполнять ли приём данных в новую таблицу ClickHouse или использовать существующую.
    Следуйте инструкциям на экране, чтобы изменить имя таблицы, схему и параметры.
    Предварительный просмотр изменений в реальном времени отображается в sample-таблице в верхней части экрана.

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_step4a.webp" alt="Задать таблицу, схему и параметры" size="lg" border />

    Вы также можете настроить дополнительные параметры с помощью доступных элементов управления

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_step4a3.webp" alt="Задать расширенные элементы управления" size="lg" border />

    Кроме того, вы можете выбрать приём данных в существующую таблицу ClickHouse.
    В этом случае интерфейс позволит сопоставить поля из источника с полями ClickHouse в выбранной целевой таблице.

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_step4b.webp" alt="Использовать существующую таблицу" size="lg" border />

    <Info>
      Вы также можете сопоставлять [virtual columns](/ru/reference/functions/table-functions/s3#virtual-columns), такие как `_path` или `_size`, с полями.
    </Info>
  </Step>

  <Step>
    ## Настройте разрешения

    Наконец, вы можете настроить разрешения для внутреннего пользователя ClickPipes.

    **Разрешения:** ClickPipes создаст отдельного пользователя для записи данных в целевую таблицу. Для этого внутреннего пользователя можно выбрать пользовательскую роль или одну из предопределённых ролей:

    * `Full access`: полный доступ к кластеру. Требуется, если вы используете materialized view или словарь с целевой таблицей.
    * `Only destination table`: только разрешения `INSERT` для целевой таблицы.

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_step5.webp" alt="Разрешения" size="lg" border />
  </Step>

  <Step>
    ## Завершите настройку

    Нажав &quot;Complete Setup&quot;, вы зарегистрируете ClickPipe, и он появится в сводной таблице.

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_success.webp" alt="Уведомление об успешном завершении" size="sm" border />

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_remove.webp" alt="Уведомление об удалении" size="lg" border />

    В сводной таблице доступны элементы управления для просмотра sample-данных из исходной или целевой таблицы в ClickHouse

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_destination.webp" alt="Просмотр пункта назначения" size="lg" border />

    А также элементы управления для удаления ClickPipe и просмотра сводки по задаче приёма.

    <Image img="/images/integrations/data-ingestion/clickpipes/cp_overview.webp" alt="Просмотр обзора" size="lg" border />

    **Поздравляем!** Вы успешно настроили свой первый ClickPipe.
    Если этот ClickPipe настроен для непрерывного приёма, он будет работать постоянно и выполнять приём данных в реальном времени из удалённого источника данных.
    В противном случае он обработает батч и завершит работу.
  </Step>
</Steps>
