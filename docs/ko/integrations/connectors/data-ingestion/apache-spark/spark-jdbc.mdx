    ```java
    public static void main(String[] args) {

                    .format("jdbc")
                    .option("url", jdbcURL)
                    .option("user", "default")
                    .option("password", "123456")
                    .option("query", query)
                    .load();

            df2.show();

            // Spark м„ём…ҳ мӨ‘м§Җ
            spark.stop();
        }
    ```

    ```java
     public static void main(String[] args) {
            // Spark sessionмқ„ мҙҲкё°нҷ”н•©лӢҲлӢӨ
            SparkSession spark = SparkSession.builder().appName("example").master("local").getOrCreate();

            // JDBC м—°кІ° м •ліҙлҘј м„Өм •н•©лӢҲлӢӨ
            String jdbcUrl = "jdbc:ch://localhost:8123/default";
            Properties jdbcProperties = new Properties();
            jdbcProperties.put("user", "default");
            jdbcProperties.put("password", "123456");

            // мҳҲмӢңмҡ© DataFrameмқ„ мғқм„ұн•©лӢҲлӢӨ
            StructType schema = new StructType(new StructField[]{
                    DataTypes.createStructField("id", DataTypes.IntegerType, false),
                    DataTypes.createStructField("name", DataTypes.StringType, false)
            });

            List<Row> rows = new ArrayList<Row>();
            rows.add(RowFactory.create(1, "John"));
            rows.add(RowFactory.create(2, "Doe"));

            Dataset<Row> df = spark.createDataFrame(rows, schema);

            //---------------------------------------------------------------------------------------------------
            // jdbc л©”м„ңл“ңлҘј мӮ¬мҡ©н•ҙ dfлҘј ClickHouseм—җ м ҖмһҘн•©лӢҲлӢӨ
            //---------------------------------------------------------------------------------------------------

            df.write()
                    .mode(SaveMode.Append)
                    .jdbc(jdbcUrl, "example_table", jdbcProperties);

            //---------------------------------------------------------------------------------------------------
            // save л©”м„ңл“ңлҘј мӮ¬мҡ©н•ҙ dfлҘј ClickHouseм—җ м ҖмһҘн•©лӢҲлӢӨ
            //---------------------------------------------------------------------------------------------------

            df.write()
                    .format("jdbc")
                    .mode("append")
                    .option("url", jdbcUrl)
                    .option("dbtable", "example_table")
                    .option("user", "default")
                    .option("password", "123456")
                    .save();

            // Spark sessionмқ„ мӨ‘м§Җн•©лӢҲлӢӨ
            spark.stop();
        }
    ```

Spark JDBCлҘј мӮ¬мҡ©н•  л•Ң SparkлҠ” лӢЁмқј нҢҢнӢ°м…ҳмңјлЎң лҚ°мқҙн„°лҘј мқҪмҠөлӢҲлӢӨ. лҚ” лҶ’мқҖ лҸҷмӢңм„ұмқ„ нҷ•ліҙн•ҳл Өл©ҙ

м—¬лҹ¬ мӣҢм»Өк°Җ лі‘л ¬лЎң мқҪмқ„ л•Ң н…Ңмқҙлё”мқ„ м–ҙл–»кІҢ нҢҢнӢ°м…ҳн• м§Җ м •мқҳн•©лӢҲлӢӨ.
